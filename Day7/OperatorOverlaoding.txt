class MyClass:
    def __init__(self,x):
        self.x=x
        print("value of x",x)
    def __lt__(self,obj):
        return self.x < obj.x
    def __gt__(self,obj):
        return self.x > obj.x
    def __eq__(self,obj):
        return (self.x == obj.x)
m1= MyClass(7)
m2= MyClass(10)
m3= MyClass(15)
m4 = m1 < m2 and m2 == m3
